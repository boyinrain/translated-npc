	end;
OnTouch:
	mes "Untuk menggunakan kapal udara, Anda diminta untuk membayar 1.200 zeny atau tiket gratis kapal udara.";
	mes "Apakah Anda ingin menggunakan layanan ini?";
	next;
	if (select("Yes:No") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Tiket_Gratis_Kapal_Udara
			warp "airplane_01",245,60;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane_01",245,60;
			end;
		}
		mes "Saya minta maaf, tetapi Anda tidak punya cukup uang.";
		mes "Harap diingat, Anda diharuskan untuk membayar 1.200 zeny untuk menggunakan layanan ini.";
		close;
	}
	mes "Terima kasih, silakan datang lagi.";
	close;
}
